::title::
Términos clave

::content::

* **Activo de información:** dato, sistema, documento, proceso o recurso que tiene valor para la organización.
* **Amenaza:** evento potencial que puede causar daño.
* **Vulnerabilidad:** debilidad que puede ser explotada por una amenaza.
* **Riesgo:** combinación entre probabilidad e impacto.
* **Control:** medida administrativa, técnica o física que reduce el riesgo.
* **Objetivo de seguridad:** resultado esperado que protege la información y apoya la estrategia.
* **Mejora continua:** revisión y ajuste periódico del SGSI.

<!--
Notas del presentador:
Usar un ejemplo sencillo: una plataforma académica es un activo; el phishing es una amenaza; una contraseña débil es una vulnerabilidad; el acceso no autorizado es un riesgo; la autenticación multifactor es un control; reducir accesos indebidos puede ser un objetivo de seguridad.
-->

```python {lines:true}
# Práctica guiada: Priorización básica de vulnerabilidades para un SGSI
# Curso: Gestión de la Seguridad de la Información
# Semana 3: Alineación estratégica del SGSI y controles de vulnerabilidad

from dataclasses import dataclass
from typing import List


@dataclass
class Vulnerabilidad:
    proceso: str
    activo: str
    descripcion: str
    probabilidad: int
    impacto: int
    control: str
    objetivo_estrategico: str

    def puntaje_riesgo(self) -> int:
        return self.probabilidad * self.impacto

    def nivel_riesgo(self) -> str:
        puntaje = self.puntaje_riesgo()

        if puntaje >= 20:
            return "Crítico"
        elif puntaje >= 12:
            return "Alto"
        elif puntaje >= 6:
            return "Medio"
        else:
            return "Bajo"


def validar_escala(valor: int, nombre: str) -> None:
    if valor < 1 or valor > 5:
        raise ValueError(f"{nombre} debe estar entre 1 y 5.")


def crear_vulnerabilidad(
    proceso: str,
    activo: str,
    descripcion: str,
    probabilidad: int,
    impacto: int,
    control: str,
    objetivo_estrategico: str
) -> Vulnerabilidad:
    validar_escala(probabilidad, "La probabilidad")
    validar_escala(impacto, "El impacto")

    return Vulnerabilidad(
        proceso=proceso,
        activo=activo,
        descripcion=descripcion,
        probabilidad=probabilidad,
        impacto=impacto,
        control=control,
        objetivo_estrategico=objetivo_estrategico
    )


def ordenar_por_riesgo(vulnerabilidades: List[Vulnerabilidad]) -> List[Vulnerabilidad]:
    return sorted(
        vulnerabilidades,
        key=lambda item: item.puntaje_riesgo(),
        reverse=True
    )


def mostrar_reporte(vulnerabilidades: List[Vulnerabilidad]) -> None:
    print("REPORTE DE PRIORIZACIÓN DE VULNERABILIDADES DEL SGSI")
    print("=" * 70)

    for indice, item in enumerate(ordenar_por_riesgo(vulnerabilidades), start=1):
        print(f"\nCaso {indice}")
        print("-" * 70)
        print(f"Proceso de negocio: {item.proceso}")
        print(f"Activo de información: {item.activo}")
        print(f"Vulnerabilidad: {item.descripcion}")
        print(f"Probabilidad: {item.probabilidad}")
        print(f"Impacto: {item.impacto}")
        print(f"Puntaje de riesgo: {item.puntaje_riesgo()}")
        print(f"Nivel de riesgo: {item.nivel_riesgo()}")
        print(f"Control sugerido: {item.control}")
        print(f"Objetivo estratégico afectado: {item.objetivo_estrategico}")


def resumen_por_nivel(vulnerabilidades: List[Vulnerabilidad]) -> None:
    conteo = {
        "Crítico": 0,
        "Alto": 0,
        "Medio": 0,
        "Bajo": 0
    }

    for item in vulnerabilidades:
        conteo[item.nivel_riesgo()] += 1

    print("\nRESUMEN POR NIVEL DE RIESGO")
    print("=" * 70)

    for nivel, cantidad in conteo.items():
        print(f"{nivel}: {cantidad}")


def recomendar_accion(nivel: str) -> str:
    acciones = {
        "Crítico": "Atención inmediata, asignación de recursos y seguimiento directivo.",
        "Alto": "Plan de tratamiento priorizado y responsable definido.",
        "Medio": "Monitoreo, control preventivo y revisión periódica.",
        "Bajo": "Aceptación controlada o seguimiento básico."
    }

    return acciones.get(nivel, "Nivel no reconocido.")


def mostrar_plan_accion(vulnerabilidades: List[Vulnerabilidad]) -> None:
    print("\nPLAN DE ACCIÓN SUGERIDO")
    print("=" * 70)

    for item in ordenar_por_riesgo(vulnerabilidades):
        print(f"\nActivo: {item.activo}")
        print(f"Nivel: {item.nivel_riesgo()}")
        print(f"Acción recomendada: {recomendar_accion(item.nivel_riesgo())}")


def main() -> None:
    vulnerabilidades = [
        crear_vulnerabilidad(
            proceso="Gestión académica",
            activo="Plataforma LMS",
            descripcion="Contraseñas débiles en cuentas de usuarios",
            probabilidad=4,
            impacto=5,
```
